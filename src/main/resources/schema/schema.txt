CREATE TABLE YUVENTORY.ITEMS (
    ID INTEGER GENERATED ALWAYS AS IDENTITY NOT NULL,
    NAME VARCHAR(100),
    ITEM_TYPE VARCHAR(30),
    WEIGHT DOUBLE,
    DATE TIMESTAMP,
    STOCK_TYPE INTEGER,
    USER_ENTERED INTEGER,
    SUPPLIER INTEGER,
    PRIMARY KEY (ID)
);

CREATE TABLE YUVENTORY.STOCK_TYPE (
    ID INTEGER GENERATED ALWAYS AS IDENTITY NOT NULL,
    NAME VARCHAR(30),
    PRIMARY KEY (ID)
);

CREATE TABLE YUVENTORY.SUPPLIER (
    ID INTEGER GENERATED ALWAYS AS IDENTITY NOT NULL,
    NAME VARCHAR(50),
    PRIMARY KEY (ID)
);

CREATE TABLE YUVENTORY.USERS (
    ID INTEGER GENERATED ALWAYS AS IDENTITY NOT NULL,
    USER_ID VARCHAR(25),
    PASSWORD VARCHAR(25),
    USER_NAME VARCHAR(25),
    ENABLED BOOLEAN(1),
    PRIMARY KEY (ID)
);
